\begin{frame}{Formulação em Reconhecimento de Padrões}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.50\textwidth}
            \large Cada pixel ou pequena vizinhança é tratado como uma amostra.
            \vspace{0.25cm}

            \scriptsize
            \[
                x \rightarrow
                \phi(x)=
                [c_{rgb}, c_{hsv}, l_{lab}, \sigma_{local}, s_{linear}]
            \]
            \[
                \omega \in \{fundo, asfalto, terra\}
            \]

            \begin{itemize}
                \item \(x\): padrão observado na imagem;
                \item \(\phi(x)\): vetor de características;
                \item \(\omega\): classe estimada pelo sistema.
            \end{itemize}
        \end{column}
        \begin{column}{0.45\textwidth}
            \begin{center}
                \begin{tikzpicture}[node distance=0.2cm]
                    \node[draw=RoadBlue!45,fill=RoadBlue!9,rounded corners=3pt,minimum width=3.1cm,minimum height=0.72cm,align=center] (a) {\scriptsize amostra};
                    \node[draw=RoadBlue!45,fill=RoadBlue!9,rounded corners=3pt,minimum width=3.1cm,minimum height=0.72cm,align=center,below=of a] (b) {\scriptsize atributos};
                    \node[draw=RoadOrange!45,fill=RoadOrange!9,rounded corners=3pt,minimum width=3.1cm,minimum height=0.72cm,align=center,below=of b] (c) {\scriptsize decisão};
                    \node[draw=SignalRed!45,fill=SignalRed!9,rounded corners=3pt,minimum width=3.1cm,minimum height=0.72cm,align=center,below=of c] (d) {\scriptsize classe};
                    \draw[->,line width=1.3pt,Ink!45] (a) -- (b);
                    \draw[->,line width=1.3pt,Ink!45] (b) -- (c);
                    \draw[->,line width=1.3pt,Ink!45] (c) -- (d);
                \end{tikzpicture}
            \end{center}
            \vspace{0.25cm}

            \small
            A abordagem implementada é explicável: usa regras determinísticas em vez de treinamento supervisionado, porque a entrega não possui máscaras anotadas.
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Classificador por Regras}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.48\textwidth}
            \large A decisão combina conhecimento de domínio e limiares adaptados à imagem.
            \vspace{0.25cm}

            \scriptsize
            \begin{align*}
                M &= A \cup (T \cap L) \\
                A &= sat_{baixa} \cap cinza \cap homog \cap visivel \\
                T &= matiz_{quente} \cap solo_{rgb} \cap homog \cap visivel
            \end{align*}

            \small
            \begin{itemize}
                \item \(A\): hipótese de asfalto;
                \item \(T\): hipótese de terra;
                \item \(L\): suporte linear exigido para terra.
            \end{itemize}
        \end{column}
        \begin{column}{0.47\textwidth}
            \resultimage{demo_sintetico_mascara.png}{height=0.64\textheight}
        \end{column}

\begin{frame}{Suporte Linear e Invariância}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.48\textwidth}
            \large Vias são padrões alongados, não apenas regiões de uma cor específica.
            \vspace{0.25cm}

            \small
            \begin{itemize}
                \item Canny detecta bordas locais;
                \item kernels lineares cobrem 0° a 150°;
                \item Hessiana multi-escala destaca cristas;
                \item solo de terra só entra se houver \(L > 0{,}15\).
            \end{itemize}
        \end{column}

\begin{frame}{Segmentação e Morfologia}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.58\textwidth}
            \resultimage{demo_sintetico_sobreposicao.png}{height=0.72\textheight}
        \end{column}
        \begin{column}{0.38\textwidth}
            \large A classificação pixel a pixel precisa de coerência espacial.
            \vspace{0.22cm}

            \small
            \begin{itemize}
                \item fechamento conecta pequenas quebras;
                \item abertura remove ruído isolado;
                \item componentes pequenos são descartados;
                \item regiões compactas demais são penalizadas;
                \item a sobreposição permite inspeção visual.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Reconhecimento Estrutural}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.50\textwidth}
            \resultimage{demo_sintetico_esqueleto.png}{height=0.68\textheight}
        \end{column}
        \begin{column}{0.46\textwidth}
            \large Depois da segmentação, o padrão deixa de ser só pixel e vira estrutura.
            \vspace{0.25cm}

            \small
            \[
                grau(p)=\sum_{q\in N_8(p)} S(q)
            \]
            \begin{itemize}
                \item \(S\): esqueleto binário;
                \item grau 1: extremidade;
                \item grau 2: ponto intermediário;
                \item grau \(\geq 3\): interseção;
                \item agrupamento por componentes define os nós.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

        \begin{column}{0.38\textwidth}
            \large \(G=(V,E)\)
            \vspace{0.2cm}

            \small
            \begin{itemize}
                \item \(V\): extremidades e interseções;
                \item \(E\): caminhos do esqueleto entre vértices;
                \item peso: comprimento em pixels;
                \item geometria: coordenadas \((x,y)\);
                \item saída: maior componente conectado.
            \end{itemize}
