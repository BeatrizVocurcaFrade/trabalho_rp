\begin{frame}{Regra de Segmentação}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.48\textwidth}
            \resultimage{demo_sintetico_segmentada.png}{height=0.66\textheight}
        \end{column}
        \begin{column}{0.48\textwidth}
            \large A máscara final combina cor, textura e linearidade.
            \vspace{0.2cm}

            \scriptsize
            \begin{align*}
                M &= A \cup (T \cap L) \\
                A &= sat_{baixa} \cap cinza \cap homog \cap visivel \\
                T &= matiz_{quente} \cap solo_{rgb} \cap homog \cap visivel
            \end{align*}

            \vspace{-0.05cm}
            \small
            \begin{itemize}
                \item \(L\): suporte linear por bordas/cristas;
                \item vegetação: excesso relativo de verde;
                \item sombra: baixo valor no canal HSV.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Suporte Linear}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.48\textwidth}
            \Large A via deve parecer uma estrutura alongada.
            \vspace{0.35cm}

            \small
            \begin{itemize}
                \item Canny detecta bordas locais;
                \item kernels lineares dilatam bordas em 6 orientações;
                \item resposta Hessiana multi-escala destaca cristas;
                \item solo de terra só entra se houver \(L > 0{,}15\).
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}

\begin{frame}{Pós-processamento}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.58\textwidth}
            \resultimage{demo_sintetico_sobreposicao.png}{height=0.72\textheight}
        \end{column}
        \begin{column}{0.38\textwidth}
            \large A máscara bruta é refinada antes da topologia.
            \vspace{0.22cm}

            \small
            \begin{itemize}
                \item fechamento conecta pequenas quebras;
                \item abertura remove ruído isolado;
                \item componentes pequenos são descartados;
                \item blobs compactos grandes são penalizados;
                \item regiões alongadas são preservadas.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Esqueleto}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.50\textwidth}
            \resultimage{demo_sintetico_esqueleto.png}{height=0.68\textheight}
        \end{column}
        \begin{column}{0.46\textwidth}
            \large A máscara vira uma estrutura topológica.
            \vspace{0.25cm}

            \small
            \[
                grau(p)=\sum_{q\in N_8(p)} S(q)
            \]
            \begin{itemize}
                \item \(S\): esqueleto binário;
                \item grau 1: extremidade;
                \item grau 2: ponto intermediário;
                \item grau \(\geq 3\): interseção;
                \item componentes de nós são agrupados por centróide.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Construção do Grafo}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.58\textwidth}
            \resultimage{demo_sintetico_grafo.png}{height=0.72\textheight}
        \end{column}
        \begin{column}{0.38\textwidth}
            \large \(G=(V,E)\)
            \vspace{0.2cm}

            \small
            \begin{itemize}
                \item \(V\): extremidades e interseções;
                \item \(E\): caminhos do esqueleto entre vértices;
                \item peso: comprimento aproximado em pixels;
                \item geometria: sequência de coordenadas \((x,y)\);
                \item saída: maior componente conectado.
            \end{itemize}

            \vspace{0.1cm}
            \pill{SignalRed}{JSON + GraphML}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Limitações}
    \begin{columns}[T,totalwidth=\textwidth]
        \begin{column}{0.48\textwidth}
            \Large O método é explicável, mas não universal.
            \vspace{0.3cm}

            \begin{itemize}
                \item sem base anotada, avaliação é qualitativa;
                \item objetos parecidos com vias ainda geram ruído;
                \item o grafo depende da continuidade da máscara.
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \Large Próximos passos.
            \vspace{0.3cm}

            \begin{itemize}
                \item medir IoU, precisão, revocação e F1;
                \item testar U-Net, DeepLab ou SegFormer;
                \item comparar com OpenStreetMap;
                \item refinar conectividade topológica.
            \end{itemize}
        \end{column}
    \end{columns}
